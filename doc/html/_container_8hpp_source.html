177 template <typename T> T *resolve() {
178 KNOT_INFO("Resolving service with ID: {}", TypeId<T>());
179 RegistryEntry *entry = find_entry(TypeId<T>());
180 if (!entry)
181 return nullptr;
182 Descriptor &desc = entry->desc;
183 switch (desc.strategy) {
184 case SINGLETON: {
185 if (!desc.instance)
186 desc.instance = desc.factory->create(desc.storage);
187 return static_cast<T *>(desc.instance);
188 }
189 case TRANSIENT: {
190 if (_transient_count >= KNOT_MAX_TRANSIENTS)
191 return nullptr;
192 size_t size = 0;
193 KNOT_ERR("Allocating transient service with ID: {}", TypeId<T>());
194 void *mem = _pool.allocate(sizeof(T), sizeof(void *), &size);
195 if (!mem)
196 return nullptr;
197 T *ptr = static_cast<T *>(desc.factory->create(mem));
198 _transients[_transient_count].factory = desc.factory;
199 _transients[_transient_count].ptr = ptr;
200 _transients[_transient_count].alloc_size = size;
201 ++_transient_count;
202 return ptr;
203 }
204 default:
205 return nullptr;
206 }
207 }

225 void destroyAllTransients() {

227 if (_transients[i].ptr && _transients[i].factory)
228 _transients[i].factory->destroy(_transients[i].ptr);
229 if (_transients[i].ptr)
230 _pool.deallocate(_transients[i].ptr, _transients[i].alloc_size);
231 _transients[i].ptr = nullptr;
232 _transients[i].alloc_size = 0;
233 _transients[i].factory = nullptr;

246 template <typename T> void destroyTransient(T *ptr) {
247 for (size_t i = 0; i < _transient_count; ++i) {
248 if (_transients[i].ptr == ptr) {
249 _transients[i].factory->destroy(ptr);
250 _pool.deallocate(_transients[i].ptr, _transients[i].alloc_size);
251 _transients[i].ptr = nullptr;
252 _transients[i].alloc_size = 0;
253 _transients[i].factory = nullptr;
254 for (size_t j = i + 1; j < _transient_count; ++j)
255 _transients[j - 1] = _transients[j];
256 --_transient_count;
257 break;
258 }
259 }
